21def recuperation_page(url): 

22 """Récupère la page web, puis permet de générer la liste 

23 complète des vins et ainsi récupérer les liens href pour accéder 

24 aux caractéristiques.""" 

25 browser = webdriver.Chrome() 

26 browser.get(url) 

27 try: 

28 accept_cookies = browser.find_element(By.ID, "axeptio_btn_acceptAll") 

29 browser.execute_script("arguments[0].click();", accept_cookies) 

30 except NoSuchElementException: 

31 next 

32 for i in range(1, 20): 

33 try: 

34 accept_cookies = browser.find_element(By.ID, "axeptio_btn_acceptAll") 

35 browser.execute_script("arguments[0].click();", accept_cookies) 

36 except NoSuchElementException: 

37 next 

38 try: 

39 nxt = browser.find_element( 

40 By.XPATH, '//*[@id="content"]/div[3]/div/div[2]/div/div[2]/div/div[5]/a' 

41 ) 

42 time.sleep(6) 

43 nxt.click() 

44 i = i + 1 

45 except ElementNotInteractableException: 

46 

47 pass 

48 except StaleElementReferenceException: 

49 pass 

50 except ElementClickInterceptedException: 

51 next 

52 

53 elems = browser.find_elements(By.CSS_SELECTOR, ".product-visual [href]") 

54 links = [elem.get_attribute("href") for elem in elems] 

55 return links

58def carac(lienbis): 

59 """Permet de récupérer le stock, la couleur, l'origine, l'appélation,  

60 la capacité, le degrée, la marque et le prix""" 

61 # LE STOCK 

62 try: 

63 le_stock = lienbis.find_element( 

64 By.XPATH, 

65 './/*[@id="content"]/div[2]/div[3]/article/div[2]/div/div[2]/div[3]/div[2]/div[2]/div[2]/div/div[2]/div[1]/div/span', 

66 ).text 

67 except NoSuchElementException: 

68 le_stock == "" 

69 if le_stock == "": 

70 le_stock = "Disponible uniquement en magasin" 

71 # LA COULEUR 

72 try: 

73 la_couleur = lienbis.find_element( 

74 By.XPATH, 

75 ".//*[@id='content']/div[2]/div[3]/article/div[2]/div/div[2]/div[2]/div/div[1]/div/span[1]/div/div/div", 

76 ).text 

77 except NoSuchElementException: 

78 la_couleur = "" 

79 # L'ORIGNE 

80 l_origine = lienbis.find_element(By.CLASS_NAME, "origin").text 

81 # L'APPELATION 

82 try: 

83 l_appelation = lienbis.find_element( 

84 By.XPATH, 

85 ".//*[@id='content']/div[2]/div[3]/article/div[2]/div/div[2]/div[2]/div/div[1]/div/span[2]/div/div/div", 

86 ).text 

87 except NoSuchElementException: 

88 l_appelation = "" 

89 if l_appelation == "SANS IG": 

90 l_appelation = "Sans appélation" 

91 if l_origine == l_appelation: 

92 l_appelation = "Sans appélation" 

93 # CAPACITE, DEGREE, MARQUE ET PRIX 

94 la_capacity = lienbis.find_element(By.CLASS_NAME, "capacity").text 

95 le_degree = lienbis.find_element(By.CLASS_NAME, "degree").text 

96 le_brand = lienbis.find_element(By.CLASS_NAME, "brand").text 

97 le_prix = lienbis.find_element(By.XPATH, "(//span[@class='price-value'])[2]").text 

98 

99 dict_cara = { 

100 "DEGREE": le_degree, 

101 "MARQUE": le_brand, 

102 "CAPACITE": la_capacity, 

103 "ORIGINE": l_origine, 

104 "APPELATION": l_appelation, 

105 "COULEUR": la_couleur, 

106 "STOCK": le_stock, 

107 "PRIX": le_prix, 

108 } 

109 return dict_cara

112def carac_gout(lienbis): 

113 """Permet de récupérer les données concernant le goût c'est à dire 

114 la puissance, le fruité, la vivacité et l'épicée-boisé.""" 

115 mesure = lienbis.find_elements(By.CSS_SELECTOR, ".measurement-bar [style]") 

116 caracte = [mesures.get_attribute("style") for mesures in mesure] 

117 le_fruite = caracte[0].replace("width: ", "").replace("%;", "") 

118 la_vivacite = caracte[1].replace("width: ", "").replace("%;", "") 

119 la_puissance = caracte[2].replace("width: ", "").replace("%;", "") 

120 epice_boise = caracte[3].replace("width: ", "").replace("%;", "") 

121 

122 dict_gout = { 

123 "FRUITE": le_fruite, 

124 "VIVACITE": la_vivacite, 

125 "PUISSANCE": la_puissance, 

126 "EPICE_BOISE": epice_boise, 

127 } 

128 return dict_gout

131def carac_tec(lienbis): 

132 """Permet de récupérer les données tels que la présence de sulfites,  

133 le millésime, le carafage, cépage, le taux de sucre et le type d'agriculte.""" 

134 donnees_techniques = lienbis.find_element( 

135 By.CLASS_NAME, "product-technical-data-ctn" 

136 ) 

137 dtech = donnees_techniques.text.split("\n") 

138 dict_tr = {} 

139 for i in range(0, len(dtech)): 

140 if dtech[i] == "SULFITES": 

141 dict_tr.update({"SULFITES": dtech[i + 1]}) 

142 if dtech[i] == "MILLÉSIME": 

143 dict_tr.update({"MILLESIME": dtech[i + 1]}) 

144 if dtech[i] == "CARAFAGE": 

145 dict_tr.update({"CARAFAGE": dtech[i + 1]}) 

146 if dtech[i] == "CÉPAGE": 

147 dict_tr.update( 

148 {"CEPAGE": dtech[i + 1 :]} 

149 ) # cépage est toujours le dernier élement des Dé techniques 

150 if dtech[i] == "TAUX DE SUCRE": 

151 dict_tr.update({"TAUX_DE_SUCRE": dtech[i + 1]}) 

152 if dtech[i] == "TYPE AGRICULTURE": 

153 dict_tr.update({"TYPE_AGRICULTURE": dtech[i + 1]}) 

154 

155 if "SULFITES" not in dict_tr: 

156 dict_tr.update({"SULFITES": np.nan}) 

157 if "MILLESIME" not in dict_tr: 

158 dict_tr.update({"MILLESIME": np.nan}) 

159 if "CARAFAGE" not in dict_tr: 

160 dict_tr.update({"CARAFAGE": np.nan}) 

161 if "CEPAGE" not in dict_tr: 

162 dict_tr.update({"CEPAGE": np.nan}) 

163 if "TAUX_DE_SUCRE" not in dict_tr: 

164 dict_tr.update({"TAUX_DE_SUCRE": np.nan}) 

165 if "TYPE_AGRICULTURE" not in dict_tr: 

166 dict_tr.update({"TYPE_AGRICULTURE": np.nan}) 

167 

168 return dict_tr

171def recup_caracs(links): 

172 """permet de réunir les dictionnaires généres par les fonctions carac,  

173 carac_gout et carac_tech. Et crée un json avec ces données.""" 

174 item_dic = [] 

175 lienbis = webdriver.Chrome() 

176 for i in range(len(links) - 1): 

177 lienbis.get(links[i]) 

178 dict_cara = carac(lienbis) 

179 dict_gout = carac_gout(lienbis) 

180 dict_tr = carac_tec(lienbis) 

181 

182 ##Fusion de tous les dictionnaires 

183 dict_cara.update(dict_gout) 

184 dict_cara.update(dict_tr) 

185 dict_finale = dict_cara 

186 item_dic.append(dict_finale) 

187 with open("data_scrapping.json", "w") as fp: 

188 json.dump(item_dic, fp)
